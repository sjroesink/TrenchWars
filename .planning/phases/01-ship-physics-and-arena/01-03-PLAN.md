**1. `packages/client/src/input.ts` -- InputManager**
```typescript
export class InputManager {
  private keys: Map<string, boolean> = new Map();

  constructor() {
    window.addEventListener('keydown', (e) => this.keys.set(e.code, true));
    window.addEventListener('keyup', (e) => this.keys.set(e.code, false));
  }

  poll(): ShipInput {
    return {
      left: this.keys.get('ArrowLeft') ?? false,
      right: this.keys.get('ArrowRight') ?? false,
      thrust: this.keys.get('ArrowUp') ?? false,
      afterburner: this.keys.get('ShiftLeft') ?? this.keys.get('ShiftRight') ?? false,
    };
  }
}
```

**4. `packages/client/src/game-loop.ts` -- GameLoop**
- Fixed-timestep accumulator pattern from research:
  ```typescript
  const TICK_DT = 0.01; // 100Hz
  let accumulator = 0;
  let lastTime = performance.now();

  function loop(currentTime: number) {
    const frameTime = Math.min((currentTime - lastTime) / 1000, 0.05);
    lastTime = currentTime;
    accumulator += frameTime;

    const input = inputManager.poll();

    while (accumulator >= TICK_DT) {
      updateShipPhysics(shipState, input, shipConfig, TICK_DT);
      applyWallCollision(shipState, TICK_DT, tileMap, shipConfig.radius, DEFAULT_BOUNCE_FACTOR);
      accumulator -= TICK_DT;
    }

    camera.update(shipState.x, shipState.y, ...);
    renderer.render(shipState, camera);
    requestAnimationFrame(loop);
  }
  ```
